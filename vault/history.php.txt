<script>

let page = 1;

async function loadHistory(){

let res = await fetch('/rwa/cert/api/my-gold-packet-history.php?page='+page);

let j = await res.json();

if(!j.ok)return;

if(page === 1){

document.getElementById("summary").innerHTML =
"Total Records: "+j.summary.total_rows+
"<br>Total TON: "+j.summary.allocated_total_ton;

}

let tbody = document.getElementById("rows");

j.items.forEach(r=>{

let tr = document.createElement("tr");

tr.innerHTML = `
<td>${r.distribution_date}</td>
<td>${r.cert_count}</td>
<td>${r.allocated_ton}</td>
<td class="status-${r.status}">${r.status}</td>
<td>${r.payout_tx_hash ? r.payout_tx_hash.slice(0,12)+'...' : '-'}</td>
`;

tbody.appendChild(tr);

});

if(!j.pagination.has_next){
document.getElementById("loadmore").style.display="none";
}

page++;

}

document.getElementById("loadmore").onclick = loadHistory;

loadHistory();

</script>
